[−][src]Crate source_error

A magical type for displaying source file errors

example

use source_error::{from_file, Position};
use std::error::Error;

fn main() -> Result<(), Box<dyn Error>> {
    println!(
      "{}",
      from_file("whoopsie!", "tests/source.json", Position::new(3, 4))?
    );
    Ok(())
}
